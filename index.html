The Arpeggiator


var notes = [], step  = 0;

input.onmidimessage = function(message) {
  if(message.data[0] !== NOTE_ON) { return; }
  notes.push(message.data[1]);
}

function loop() {
  if(!notes.length) { return; }
  noteOff(notes[step]);
  
  if(step++ >= notes.length) { step = 0; }
  noteOn(notes[step]);

  setTimeout(loop, 120); // ~500bpm
};
